\begin{definitions}
	\definition{Сетевое обучение}{парадигма учебной деятельности, основой которой является открытость образовательных ресурсов, а также повсеместное сотрудничество участников образовательного процесса.}
	\definition{Кана}{японские слоговые алфавиты: хирагана и катакана.}
	\definition{Хирагана}{японский слоговой алфавит, который используется для записи слов, не имеющих кандзи.}
	\definition{Катакана}{японский слоговой алфавит, который используется для записи иностранных слов и выделения важных слов в тексте.}
	\definition{Кандзи}{китайские иероглифы, которые используются в японской письменности. Кандзи были заимствованы из китайского языка и адаптированы для использования в японском языке}
\end{definitions}
